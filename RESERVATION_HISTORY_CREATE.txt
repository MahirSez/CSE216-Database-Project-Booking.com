CREATE TABLE public.reservation_history
(
  history_id integer NOT NULL DEFAULT nextval('reservation_history_history_id_seq'::regclass),
  room_id integer NOT NULL DEFAULT nextval('reservation_history_room_id_seq'::regclass),
  client_id integer NOT NULL DEFAULT nextval('reservation_history_client_id_seq'::regclass),
  check_in date,
  check_out date,
  CONSTRAINT reservation_history_pkey PRIMARY KEY (history_id),
  CONSTRAINT reservation_history_client_id_fkey FOREIGN KEY (client_id)
      REFERENCES public.client (client_id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT reservation_history_room_id_fkey FOREIGN KEY (room_id)
      REFERENCES public.rooms (room_id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
WITH (
  OIDS=FALSE
);
